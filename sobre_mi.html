document.addEventListener('DOMContentLoaded', () => { const user = JSON.parse(localStorage.getItem('user')); if (!user || user.role !== 'visitor') { alert('Acceso no autorizado'); window.location.href = 'login.html'; return; } const eventList = document.getElementById('eventList'); async function loadEvents() { const res = await fetch('http://localhost:3000/events'); const events = await res.json(); const registrationsRes = await fetch('http://localhost:3000/registrations'); const registrations = await registrationsRes.json(); eventList.innerHTML = ''; events.forEach((event, index) => { const count = registrations.filter(r => r.eventId === event.id).length; const alreadyRegistered = registrations.some(r => r.eventId === event.id && r.userId === user.id); const remaining = event.capacity - count; eventList.innerHTML += ` ${index + 1} ${event.title} ${event.date} ${event.capacity} ${count} ${alreadyRegistered ? 'Registrado' : remaining > 0 ? `` : 'Cupo lleno'} `; }); } window.register = async (eventId) => { await fetch('http://localhost:3000/registrations', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: user.id, eventId }) }); alert("Te has registrado al evento."); loadEvents(); }; document.getElementById('logoutBtn').addEventListener('click', () => { localStorage.removeItem('user'); window.location.href = 'login.html'; }); loadEvents(); });
